package $basePackage.web;

import static net.yadaframework.core.YadaConstants.VAL_NOTIFICATION_SEVERITY_INFO;

import javax.servlet.http.HttpServletResponse;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.RequestMapping;

import net.yadaframework.web.YadaWebUtil;

@Controller
public class HomeController {
	@Autowired YadaWebUtil yadaWebUtil;

	// Pagina aperta dopo il timeout di sessione rilevato via checkSession
	@RequestMapping("/timeout")
	public String timeout(Model model) {
		YadaNotify.instance(model).yadaTitle("Session Expired")
			.yadaMessage("Session expired for inactivity, please login again")
			.yadaInfo().redirectOnClose("/").yadaSave();
		return "/home";
	}

	@RequestMapping("/")
	public String home(Model model, HttpServletResponse response) {
		return "/home";
	}
	
	// La pagina di login Ã¨ semplicemente la home con un attributo settato
	@RequestMapping("/login")
	public String login(Model model) {
		model.addAttribute("login", "login");
		return "/home";
	}
	

}
